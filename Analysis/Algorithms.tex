
	\begin{flushleft}
There are many algorithms used in selling products and controlling the stock of the product. \par

These Algorithms are used when Selling Products: \par

Adding a New Product to the Total Price:
	\end{flushleft}

\begin{algorithm}[H]
	\caption{Adding Product to Total Price}
\begin{algorithmic}[1]
\SET{$Done$}{$False$}
\SET{$TotalPrice$}{$0$}
\State
\While{$Not Done$}
	\SEND{$"Please\ enter\ the\ price\ of \ the\ product (0.00)"$}
	\RECEIVE{$Price$}
	\If{$Price == 0$}
	\SET{$Done$}{$True$}
	\Else
	\SET{$TotalPricel$}{$TotalPrice + Price$}
	\EndIf
\EndWhile
\end{algorithmic}
\end{algorithm}




	\begin{flushleft}
Changing the quantity of the product being sold:
	\end{flushleft}

\begin{algorithm}[H]
	\caption{Changing the Quantity}
\begin{algorithmic}[1]
\SET{$Done$}{$False$}
\SEND{$"Please\ enter\ the\ quantity\ of \ the\ product (0-100)"$}
\RECEIVE{$Quantity$}
\If {$Quantity = 0$}
\SET{Done$}{$True$}
\Else
\SET{$TotalPricel$}{$TotalPrice + (Price*Quantity)$}
\EndIf
\end{algorithmic}
\end{algorithm}

Calculating Change:
	\end{flushleft}	
	
\begin{algorithm}[H]
	\caption{Calculating Change}
\begin{algorithmic}[1]
\SET{$Done$}{$False$}
\SEND{$''Please Enter Money Given'')"$}
\RECEIVE{$MoneyGiven$}
\If {$MoneyGiven\ <\  Price$)
\SEND ($''More\ Money\ Please''}
\Elsif {$MoneyGiven\ =\  Price$}
\SEND ($''Thank\ you!''$)
\Elsif {$MoneyGiven\ >\  Price$}
\SET {$Change$}{$MoneyGiven - Price$}
\SEND{$Change$)
\EndIf
\EndElsIf
\EndElsif
\end{algorithmic}
\end{algorithm}
